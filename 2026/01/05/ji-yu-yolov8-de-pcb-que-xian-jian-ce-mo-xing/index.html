def convert_coordinates(size, box):
    dw = 1. / size[0]
    dh = 1. / size[1]
    x = (box[0] + box[2]) / 2.0
    y = (box[1] + box[3]) / 2.0
    w = box[2] - box[0]
    h = box[3] - box[1]
    return (x * dw, y * dh, w * dw, h * dh)

def process_list(file_list, save_stage):
    img_save_dir = os.path.join(dst_root, 'images', save_stage)
    lbl_save_dir = os.path.join(dst_root, 'labels', save_stage)
    os.makedirs(img_save_dir, exist_ok=True)
    os.makedirs(lbl_save_dir, exist_ok=True)

    missing_count = 0
    for line in file_list:
        parts = line.strip().split()
        if len(parts) < 2: continue
        
        img_rel_path, lbl_rel_path = parts[0], parts[1]
        
        # --- æ ¸å¿ƒä¿®æ­£ï¼šå¤„ç†æ–‡ä»¶ååç¼€é—®é¢˜ ---
        img_path = os.path.join(src_root, img_rel_path)

        if not os.path.exists(img_path):

                print(f"è­¦å‘Š: æ‰¾ä¸åˆ°å›¾ç‰‡ {img_path}")

        # è·å–ä¿å­˜æ–‡ä»¶å

        img = cv2.imread(img_path)
        h, w, _ = img.shape
        
        yolo_annotations = []
        with open(lbl_path, 'r') as f:

                x1, y1, x2, y2, cls_id = data

    print(f"{save_stage} å¤„ç†å®Œæˆã€‚è·³è¿‡ç¼ºå¤±æ–‡ä»¶æ•°: {missing_count}")
