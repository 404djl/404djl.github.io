Lab1：Xv6 and Unix utilities

实验环境：ubuntu22虚拟机

Boot xv6 (easy)

安装基本工具：

sudo apt-get install git build-essential gdb-multiarch qemu-system-misc gcc-riscv64-linux-gnu binutils-riscv64-linux-gnu

下载实验源码：

git clone git://g.csail.mit.edu/xv6-labs-2024

cd xv6-labs-2024

编译和启动 xv6：

make qemu

  • 进入后可以使用一些 基本的命令

    raw,id=x0 -device virtio-blk-device,drive=x0,bus=virtio-mmio-bus.0

sleep (easy)

Implement a user-level sleep program for xv6, along the lines of the UNIX sleep command. Your sleep should pause for a user-specified number of ticks. A tick is a notion of time defined by the xv6 kernel, namely the time between two interrupts from the timer chip. Your solution should be in the file user/sleep.c.

  1. 在项目的user文件夹下创建sleep.c文件
  2. 在顶层Makefile中的UPROGS变量中添加$U/_sleep
#include "kernel/types.h"
#include "user/user.h"

int main(int argc, char *argv[]) {
    if (argc != 2) {
        // 参数数量错误处理
        fprintf(2, "Error: Invalid argument count\nUsage: sleep ticks\n");
        exit(1);
    }

    // 参数有效性验证
    char *p = argv[1];
    while (*p) {
        if (*p < '0' || *p > '9') {
            fprintf(2, "Error: Non-numeric argument '%s'\n", argv[1]);
            exit(1);
        }
        p++;
    }

    // 执行睡眠操作
    sleep(atoi(argv[1]));
    exit(0);
}

pingpong (easy)

Write a user-level program that uses xv6 system calls to ‘’ping-pong’’ a byte between two processes over a pair of pipes, one for each direction. The parent should send a byte to the child; the child should print “: received ping”, where is its process ID, write the byte on the pipe to the parent, and exit; the parent should read the byte from the child, print “: received pong”, and exit. Your solution should be in the file user/pingpong.c.

  1. 在项目的user文件夹下创建pingpong.c文件
  2. 在顶层Makefile中的UPROGS变量中添加$U/_pingpong
#include "kernel/types.h"
#include "user/user.h"

int main() {
    // 创建两个管道：ping（父写子读）和 pong（子写父读）
    int ping[2], pong[2];
    pipe(ping);
    pipe(pong);

    if (fork() == 0) { 
        // 子进程处理逻辑
        close(ping[1]);  // 关闭父进程的写端
        close(pong[0]);  // 关闭父进程的读端

        char received;
        read(ping[0], &received, 1);             // 从父进程读取字节
        printf("%d: received ping\n", getpid());  // 打印子进程PID
        write(pong[1], &received, 1);             // 将字节写回父进程

        close(ping[0]);
        close(pong[1]);
        exit(0);
    } else { 
        // 父进程处理逻辑
        close(ping[0]);  // 关闭子进程的读端
        close(pong[1]);  // 关闭子进程的写端

        write(ping[1], "x", 1);                   // 发送字节给子进程
        char response;
        read(pong[0], &response, 1);              // 等待子进程回应
        printf("%d: received pong\n", getpid());  // 打印父进程PID

        close(ping[1]);
        close(pong[0]);
        exit(0);
    }
}

primes (moderate)/(hard)

Write a concurrent prime sieve program for xv6 using pipes and the design illustrated in the picture halfway down this page and the surrounding text. This idea is due to Doug McIlroy, inventor of Unix pipes. Your solution should be in the file user/primes.c.

  1. 在项目的user文件夹下创建primes.c文件
  2. 在顶层Makefile中的UPROGS变量中添加$U/_primes
#include "kernel/types.h"
#include "user/user.h"

#define MAX_NUM 280

void sieve(int read_fd) {
    int prime;
    
    // 读取第一个素数
    if (read(read_fd, &prime, sizeof(prime)) != sizeof(prime)) {
        close(read_fd);
        exit(0);
    }
    printf("prime %d\n", prime);

    int pipeline[2];
    pipe(pipeline);

    if (fork() == 0) {
        // 子进程处理下一级筛选
        close(pipeline[1]);  // 关闭写端
        close(read_fd);      // 关闭父级读端
        sieve(pipeline[0]);  // 递归处理新管道
    } else {
        // 当前进程过滤倍数
        close(pipeline[0]);  // 关闭读端
        int num;
        while (read(read_fd, &num, sizeof(num)) > 0) {
            if (num % prime != 0) {
                write(pipeline[1], &num, sizeof(num));
            }
        }
        // 清理资源并等待子进程
        close(read_fd);
        close(pipeline[1]);
        wait(0);
    }
    exit(0);
}

int main() {
    int init_pipe[2];
    pipe(init_pipe);

    if (fork() == 0) {
        // 子进程启动筛法
        close(init_pipe[1]);  // 关闭写端
        sieve(init_pipe[0]);
    } else {
        // 主进程填充初始数据
        close(init_pipe[0]);  // 关闭读端
        for (int i = 2; i <= MAX_NUM; i++) {
            write(init_pipe[1], &i, sizeof(i));
        }
        // 关闭写端触发EOF
        close(init_pipe[1]);
        wait(0);
    }
    exit(0);
}

find (moderate)

Write a simple version of the UNIX find program for xv6: find all the files in a directory tree with a specific name. Your solution should be in the file user/find.c.

  1. 在项目的user文件夹下创建find.c文件
  2. 在顶层Makefile中的UPROGS变量中添加$U/_find
#include "kernel/types.h"
#include "kernel/stat.h"
#include "user/user.h"
#include "kernel/fs.h"

// 提取路径的基名（最后一个'/'后的部分）
char* basename(char *path) {
    char *prev = 0;
    char *p = path;
    while (*p != '\0') {
        if (*p == '/') prev = p;
        p++;
    }
    return prev ? prev + 1 : path;
}

void find(char *path, char *target) {
    char buf[512], *p;
    int fd;
    struct dirent de;
    struct stat st;

    if ((fd = open(path, 0)) < 0) {
        fprintf(2, "find: cannot open %s\n", path);
        return;
    }

    if (fstat(fd, &st) < 0) {
        fprintf(2, "find: cannot stat %s\n", path);
        close(fd);
        return;
    }

    switch (st.type) {
    case T_FILE:
        // 比较文件名并输出匹配项
        if (strcmp(basename(path), target) == 0)
            printf("%s\n", path);
        break;

    case T_DIR:
        // 构造目录路径缓冲区
        if (strlen(path) + 1 + DIRSIZ + 1 > sizeof buf) {
            printf("find: path too long\n");
            break;
        }
        strcpy(buf, path);
        p = buf + strlen(buf);
        *p++ = '/';
        
        // 遍历目录条目
        while (read(fd, &de, sizeof(de)) == sizeof(de)) {
            if (de.inum == 0) continue;
            if (!strcmp(de.name, ".") || !strcmp(de.name, "..")) continue;
            
            // 构造子路径并递归查找
            memmove(p, de.name, DIRSIZ);
            p[DIRSIZ] = 0;
            find(buf, target);
        }
        break;
    }
    close(fd);
}

int main(int argc, char *argv[]) {
    if (argc != 3) {
        fprintf(2, "Usage: find <directory> <filename>\n");
        exit(1);
    }
    find(argv[1], argv[2]);
    exit(0);
}

xargs (moderate)

Write a simple version of the UNIX xargs program for xv6: its arguments describe a command to run, it reads lines from the standard input, and it runs the command for each line, appending the line to the command’s arguments. Your solution should be in the file user/xargs.c.

  1. 在项目的user文件夹下创建xargs.c文件
  2. 在顶层Makefile中的UPROGS变量中添加$U/_xargs
#include "kernel/types.h"
#include "kernel/param.h"
#include "user/user.h"

#define MAX_LINE_LENGTH 512

// 解析输入行并执行命令
void execute_command(int argc, char *argv[], char *line) {
    char *args[MAXARG] = {0};  // 参数数组
    int args_cnt = 0;

    // 1. 添加原始命令参数
    for (int i = 1; i < argc; i++) {
        args[args_cnt++] = argv[i];
    }

    // 2. 解析输入行参数
    char *p = line;
    while (*p && args_cnt < MAXARG - 1) { // 保留最后一个位置给NULL
        // 跳过空白字符
        while (*p == ' ' || *p == '\t') p++;
        if (*p == '\0') break;

        // 记录参数起始位置
        args[args_cnt++] = p;

        // 寻找参数结束位置
        while (*p != ' ' && *p != '\t' && *p != '\0') p++;
        
        // 截断参数并移动指针
        if (*p != '\0') {
            *p = '\0';
            p++;
        }
    }

    if (args_cnt == 0) return;  // 无参数不执行

    // 3. 创建子进程执行命令
    int pid = fork();
    if (pid == 0) {
        exec(args[0], args);
        fprintf(2, "xargs: exec %s failed\n", args[0]);
        exit(1);
    } else if (pid > 0) {
        wait(0);
    } else {
        fprintf(2, "xargs: fork failed\n");
    }
}

int main(int argc, char *argv[]) {
    if (argc < 2) {
        fprintf(2, "Usage: xargs <command> [args...]\n");
        exit(1);
    }

    char line[MAX_LINE_LENGTH] = {0};
    char *ptr = line;
    int remaining = sizeof(line);

    // 逐字符读取标准输入
    while (1) {
        int n = read(0, ptr, 1);
        if (n <= 0) break;

        // 处理换行符
        if (*ptr == '\n') {
            *ptr = '\0';
            execute_command(argc, argv, line);
            ptr = line;          // 重置指针
            remaining = sizeof(line);
            continue;
        }

        // 移动指针并检查缓冲区溢出
        ptr++;
        remaining--;
        if (remaining <= 0) {
            fprintf(2, "xargs: line too long\n");
            exit(1);
        }
    }

    // 处理最后一行（无换行结尾的情况）
    if (ptr != line) {
        *ptr = '\0';
        execute_command(argc, argv, line);
    }

    exit(0);
}
